import java.util.*

/**
 * @author Yananart
 */
class Solution {
    fun postorder(root: Node?): List<Int> {
        val res = ArrayList<Int>()
        if (root != null) {
            val stack = Stack<Node>()
            stack.push(root)
            while (stack.isNotEmpty()) {
                val now = stack.pop()
                res.add(now.`val`)
                for (item in now.children) {
                    stack.push(item)
                }
            }
        }
        return res.reversed()
    }
}

class Node(var `val`: Int) {
    var children: List<Node?> = listOf()
}
