class Solution {
    fun addDigits(num: Int): Int {
        var now = num
        var sum = 0
        while (now != 0 || sum >= 10) {
            if (now == 0) {
                now = sum
                sum = 0
            }
            sum += now % 10
            now /= 10
        }
        return sum
    }
}
