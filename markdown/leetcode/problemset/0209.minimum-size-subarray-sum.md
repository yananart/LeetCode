class Solution {
    fun minSubArrayLen(target: Int, nums: IntArray): Int {
        var left = 0
        var right = 0
        var res = Int.MAX_VALUE
        var now = nums[left]
        while (true) {
            if (now < target) {
                right++
                if (right >= nums.size) {
                    break
                }
                now += nums[right]
            } else {
                if (res > right - left + 1) {
                    res = right - left + 1
                }
                now -= nums[left]
                left++
            }
        }
        return if (res == Int.MAX_VALUE) 0 else res
    }
}
