SELECT customer_id,
       COUNT(*) AS count_no_trans
FROM (SELECT customer_id
      FROM Visits AS v
               LEFT JOIN Transactions AS t ON v.visit_id = t.visit_id
      WHERE transaction_id IS NULL) AS temp
GROUP BY customer_id;
