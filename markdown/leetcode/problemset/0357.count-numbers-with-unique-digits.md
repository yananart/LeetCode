class Solution {
    companion object {
        private val dp = IntArray(10)

        init {
            dp[1] = 10
            var now = 9
            for (i in 2..8) {
                now *= (11 - i)
                dp[i] = dp[i - 1] + now
            }
        }
    }

    fun countNumbersWithUniqueDigits(n: Int): Int {
        return dp[n]
    }
}
