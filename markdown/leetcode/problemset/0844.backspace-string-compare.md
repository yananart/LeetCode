import java.util.Stack

/**
 * @author Yananart
 */
class Solution {
    fun backspaceCompare(s: String, t: String): Boolean {
        return opStr(s) == opStr(t)
    }

    private fun opStr(s: String): Stack<Char> {
        val stack = Stack<Char>()
        for (ch in s.toCharArray()) {
            if (ch == '#') {
                if (stack.isNotEmpty()) {
                    stack.pop()
                }
            } else {
                stack.push(ch)
            }
        }
        return stack
    }
}
