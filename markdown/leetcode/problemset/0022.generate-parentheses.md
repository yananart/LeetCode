class Solution {
    fun generateParenthesis(n: Int): List<String> {
        val res = ArrayList<String>()
        dfs("", n, n, res)
        return res
    }

    private fun dfs(now: String, left: Int, right: Int, res: ArrayList<String>) {
        if (left == 0) {
            if (right == 0) {
                res.add(now)
            } else {
                dfs("$now)", left, right - 1, res)
            }
        } else {
            dfs("$now(", left - 1, right, res)
            if (left < right) {
                dfs("$now)", left, right - 1, res)
            }
        }
    }
}
