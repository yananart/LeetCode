class Solution {
    fun reachingPoints(sx: Int, sy: Int, tx: Int, ty: Int): Boolean {
        var ty = ty
        var tx = tx
        while (sx < tx && sy < ty) {
            if (tx < ty) ty %= tx
            else tx %= ty
        }
        if (tx < sx || ty < sy) return false
        return if (sx == tx) (ty - sy) % tx == 0 else (tx - sx) % ty == 0
    }
}
