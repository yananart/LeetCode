class Solution {
    fun lengthOfLongestSubstring(s: String): Int {
        var maxLength = 0
        var temp = ""
        for (index in s.indices) {
            val char = s[index]
            if (temp.contains(char)) {
                if (temp.length > maxLength) {
                    maxLength = temp.length
                }
                temp = temp.substringAfter(char)
            }
            temp += char.toString()
        }
        if (temp.length > maxLength) {
            maxLength = temp.length
        }
        return maxLength
    }
}
