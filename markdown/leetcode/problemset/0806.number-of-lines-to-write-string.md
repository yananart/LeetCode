class Solution {
    fun numberOfLines(widths: IntArray, s: String): IntArray {
        var lines = 1
        var last = 0
        for (ch in s.toCharArray()) {
            val now = widths[ch - 'a']
            if (now + last > 100) {
                lines++
                last = now
            } else {
                last += now
            }
        }
        return intArrayOf(lines, last)
    }
}
