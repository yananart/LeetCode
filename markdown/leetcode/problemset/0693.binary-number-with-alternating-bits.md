class Solution {
    fun hasAlternatingBits(n: Int): Boolean {
        var now = n shr 1
        var last = n and 1
        while (now != 0) {
            last = 1 - last
            if (now and 1 != last) return false
            now = now shr 1
        }
        return true
    }
}
