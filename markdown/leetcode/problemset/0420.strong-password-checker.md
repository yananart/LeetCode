import kotlin.math.max
import kotlin.math.min

/**
 * @author Yananart
 */
class Solution {
    fun strongPasswordChecker(password: String): Int {
        val chars = password.toCharArray()
        val length = chars.size
        var haveLow = 0
        var haveNum = 0
        var haveUp = 0
        for (c in chars) {
            when (c) {
                in 'a'..'z' -> haveLow = 1
                in '0'..'9' -> haveNum = 1
                in 'A'..'Z' -> haveUp = 1
            }
        }
        val types = haveLow + haveNum + haveUp
        return if (length < 6) {
            max(6 - length, 3 - types)
        } else if (length <= 20) {
            var tot = 0
            var i = 0
            while (i < length) {
                var j = i
                while (j < length && chars[j] == chars[i]) j++
                val cnt = j - i
                if (cnt >= 3) tot += cnt / 3
                i = j
            }
            max(tot, 3 - types)
        } else {
            var tot = 0
            val cnts = IntArray(3)
            run {
                var i = 0
                while (i < length) {
                    var j = i
                    while (j < length && chars[j] == chars[i]) j++
                    val cnt = j - i
                    if (cnt >= 3) {
                        tot += cnt / 3
                        cnts[cnt % 3]++
                    }
                    i = j
                }
            }
            val base = length - 20
            var cur = base
            for (i in 0..2) {
                if (i == 2) cnts[i] = tot
                if (cnts[i] != 0 && cur != 0) {
                    val t = min(cnts[i] * (i + 1), cur)
                    cur -= t
                    tot -= t / (i + 1)
                }
            }
            base + max(tot, 3 - types)
        }
    }
}
