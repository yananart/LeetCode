class Solution {
    fun missingRolls(rolls: IntArray, mean: Int, n: Int): IntArray {
        val sum = rolls.sum()
        val target = mean * (n + rolls.size)
        var miss = target - sum
        if (miss < n || miss > 6 * n) {
            return intArrayOf()
        }
        val res = IntArray(n) { miss / n }
        miss -= res.sum()
        for (i in 0 until miss) {
            res[i] += 1
        }
        return res
    }
}
