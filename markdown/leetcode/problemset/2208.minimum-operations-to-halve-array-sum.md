import java.util.PriorityQueue

class Solution {
    fun halveArray(nums: IntArray): Int {
        // ä¼˜å…ˆé˜Ÿåˆ—ï¼Œå¤§çš„å…ˆå–å‡ºæ¥
        val queue = PriorityQueue<Double>(nums.size) { o1, o2 -> o2.compareTo(o1) }
        // ä¸€åŠçš„å€¼
        var half = 0.0
        for (item in nums) {
            half += item.toDouble() / 2
            queue.add(item.toDouble())
        }
        // å½“å‰å‡å»çš„å€¼
        var now = 0.0
        // ä½¿ç”¨çš„æ­¥æ•°
        var step = 0
        while (now < half) {
            var max = queue.poll()
            max /= 2
            now += max
            queue.add(max)
            step++
        }
        return step
    }
}
