class Solution {
    fun reverseBits(n: Int): Int {
        var res = 0
        var now = n
        repeat(32) {
            res = res.shl(1)
            if (now and 1 == 1) {
                res = res or 1
            }
            now = now.shr(1)
        }
        return res
    }
}
