class Solution {
    fun rangeBitwiseAnd(left: Int, right: Int): Int {
        var left = left
        var right = right
        var move = 0
        while (left != right) {
            left = left shr 1
            right = right shr 1
            move++
        }
        return left shl move
    }
}
