class Solution {
    fun triangularSum(nums: IntArray): Int {
        var length = nums.size
        while (length > 1) {
            for (i in 0 until length - 1) {
                nums[i] = (nums[i] + nums[i + 1]) % 10
            }
            length--
        }
        return nums[0]
    }
}
