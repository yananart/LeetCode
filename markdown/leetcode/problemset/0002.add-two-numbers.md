class Solution {
    fun addTwoNumbers(l1: ListNode?, l2: ListNode?): ListNode {
        var ll1 = l1
        var ll2 = l2
        var result = ListNode(0)
        val head = result
        var flag = 0
        while (!(ll1 == null && ll2 == null && flag == 0)) {
            if (ll1 != null) {
                flag += ll1.`val`
                ll1 = ll1.next
            }
            if (ll2 != null) {
                flag += ll2.`val`
                ll2 = ll2.next
            }
            val temp = ListNode(flag % 10)
            result.next = temp
            result = result.next!!
            flag /= 10
        }
        return head.next
    }
}
