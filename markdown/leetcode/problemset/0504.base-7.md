import kotlin.math.abs

class Solution {
    fun convertToBase7(num: Int): String {
        // æœ€å¤šåˆ°7çš„9æ¬¡æ–¹
        val neg = num < 0
        var now = abs(num)
        var ans = ""
        for (i in 1..9) {
            ans += ((now % 7).toString())
            val next = now / 7
            if (next == 0) break
            now = next
        }
        ans = ans.reversed()
        return if (neg) "-$ans" else ans
    }
}
