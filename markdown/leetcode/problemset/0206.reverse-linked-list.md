import java.util.Stack

/**
 * @author Yananart
 */
class Solution {
    fun reverseList(head: ListNode?): ListNode? {
        if (head == null) {
            return null
        }
        val nums = Stack<Int>()
        var now = head
        while (now != null) {
            nums.add(now.`val`)
            now = now.next
        }
        val res = ListNode(-1)
        now = res
        while (nums.isNotEmpty()) {
            now!!.next = ListNode(nums.pop())
            now = now.next
        }
        return res.next
    }
}
