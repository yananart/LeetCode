class Solution {
    fun divideArray(nums: IntArray): Boolean {
        val hash = HashMap<Int, Int>()
        for (item in nums) {
            if (!hash.containsKey(item)) {
                hash[item] = 0
            }
            hash[item] = hash[item]!! + 1
        }
        for ((_, v) in hash) {
            if (v % 2 != 0) {
                return false
            }
        }
        return true
    }
}
