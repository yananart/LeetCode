`dp[i]=nums[i]+max(dp[i-2], dp[i-3])`

```kotlin
import kotlin.math.max

/**
 * @author Yananart
 */
class Solution {
    fun rob(nums: IntArray): Int {
        var max = 0
        val dp = IntArray(nums.size)
        var now: Int
        for (index in nums.indices) {
            now = if (index < 2) {
                nums[index]
            } else if (index == 2) {
                nums[index] + dp[0]
            } else {
                nums[index] + max(dp[index - 2], dp[index - 3])
            }
            dp[index] = now
            if (now > max) {
                max = now
            }
        }
        return max
    }
}
```
