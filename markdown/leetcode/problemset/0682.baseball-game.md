import java.util.Stack

/**
 * @author Yananart
 */
class Solution {
    fun calPoints(ops: Array<String>): Int {
        val stack = Stack<Int>()
        for (item in ops) {
            when (item) {
                "C" -> stack.pop()
                "D" -> stack.push(2 * stack.peek())
                "+" -> stack.push(stack.peek() + stack.elementAt(stack.size - 2))
                else -> stack.push(item.toInt())
            }
        }
        return stack.sum()
    }
}
