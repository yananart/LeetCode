import kotlin.math.max

/**
 * @author Yananart
 */
class Solution {
    companion object {
        private val dp = IntArray(60)

        init {
            dp[1] = 1
            dp[2] = 1
            for (i in 3..58) {
                var max = Int.MIN_VALUE
                for (j in 1 until i) {
                    max = max(max, dp[j] * (i - j))
                    max = max(max, j * (i - j))
                }
                dp[i] = max
            }
        }
    }

    fun integerBreak(n: Int): Int {
        return dp[n]
    }
}
