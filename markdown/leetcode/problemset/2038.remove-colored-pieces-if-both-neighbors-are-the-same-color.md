class Solution {
    fun winnerOfGame(colors: String): Boolean {
        var aWins = 0
        var bWins = 0
        for (index in 1 until colors.length - 1) {
            if (colors[index] == 'A') {
                if (colors[index - 1] == colors[index + 1] && colors[index - 1] == 'A') {
                    aWins++
                }
            } else {
                if (colors[index - 1] == colors[index + 1] && colors[index - 1] == 'B') {
                    bWins++
                }
            }
        }
        return aWins > bWins
    }
}
