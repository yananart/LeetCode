class Solution {
    fun isPowerOfTwo(n: Int): Boolean {
        if (n <= 0) {
            return false
        }
        var now = n
        while (now.and(1) != 1) {
            now = now.shr(1)
        }
        return now.shr(1) == 0
    }
}

class Solution {
    fun hammingWeight(n: Int): Int {
        var res = 0
        var now = n
        var time = 0
        while (time < 32) {
            if (now == 0) {
                break
            }
            if (now.and(1) == 1) {
                res++
            }
            now = now.shr(1)
            time++
        }
        return res
    }
}
