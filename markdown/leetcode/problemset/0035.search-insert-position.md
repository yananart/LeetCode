class Solution {
    fun searchInsert(nums: IntArray, target: Int): Int {
        if (target < nums.first()) {
            return 0
        }
        if (target > nums.last()) {
            return nums.size
        }
        var left = 0
        var right = nums.size - 1
        while (true) {
            if (left == right) {
                return if (nums[left] >= target) {
                    left
                } else {
                    left + 1
                }
            }
            val mid = (left + right) / 2
            if (nums[mid] == target) {
                return mid
            } else if (nums[mid] > target) {
                right = mid
            } else {
                left = mid + 1
            }
        }
    }
}

/**

class Solution {
    fun searchInsert(nums: IntArray, target: Int): Int {
        if (target < nums.first()) {
            return 0
        }
        if (target > nums.last()) {
            return nums.size
        }
        var left = 0
        var right = nums.size - 1
        while (left <= right) {
            val mid = (left + right) / 2
            when {
                nums[mid] == target -> {
                    return mid
                }
                nums[mid] > target -> {
                    right = mid - 1
                }
                nums[mid] < target -> {
                    left = mid + 1
                }
            }
        }
        return left
    }
}
