class Solution {
    fun singleNumber(nums: IntArray): Int {
        val set = HashSet<Int>()
        for (item in nums) {
            if (set.contains(item)) {
                set.remove(item)
            } else {
                set.add(item)
            }
        }
        return set.single()
    }
}

/**

class Solution {
    fun singleNumber(nums: IntArray): Int {
        var res = 0
        for (item in nums) {
            res = res.xor(item)
        }
        return res
    }
}
