import java.util.Stack

/**
 * @author Yananart
 */
class Solution {
    fun isValid(s: String): Boolean {
        val stack = Stack<Char>()
        val brackets = mapOf('(' to ')', '{' to '}', '[' to ']')
        for (ch in s.toCharArray()) {
            when (ch) {
                '(', '{', '[' -> {
                    stack.push(ch)
                }
                ')', '}', ']' -> {
                    if (stack.isEmpty() || brackets[stack.pop()] != ch) {
                        return false
                    }
                }
            }
        }
        return stack.isEmpty()
    }
}
