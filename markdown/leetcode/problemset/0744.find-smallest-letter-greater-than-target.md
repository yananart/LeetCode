import kotlin.math.min

/**
 * @author Yananart
 */
class Solution {
    fun nextGreatestLetter(letters: CharArray, target: Char): Char {
        var lowest = letters[0]
        var far = Int.MAX_VALUE
        for (ch in letters) {
            if (ch < lowest) {
                lowest = ch
            }
            if (ch > target) {
                far = min(far, ch - target)
            }
        }
        return if (far != Int.MAX_VALUE) {
            target + far
        } else {
            lowest
        }
    }
}
