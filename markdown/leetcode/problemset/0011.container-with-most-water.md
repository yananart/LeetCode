import kotlin.math.min

/**
 * @author Yananart
 */
class Solution {
    fun maxArea(height: IntArray): Int {
        var left = 0
        var right = height.size - 1
        var max = 0
        while (left < right) {
            val area = (right - left) * min(height[left], height[right])
            if (area > max) {
                max = area
            }
            if (height[left] <= height[right]) {
                left++
            } else if (height[left] > height[right]) {
                right--
            }
        }
        return max
    }
}
