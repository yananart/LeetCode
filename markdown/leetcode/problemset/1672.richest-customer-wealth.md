class Solution {
    fun maximumWealth(accounts: Array<IntArray>) =
    IntArray(accounts.size) {
        var sum = 0
        for (item in accounts[it]) {
            sum += item
        }
        sum
    }.let {
        var max = 0
        for (item in it) {
            max = kotlin.math.max(item, max)
        }
        max
    }
}
