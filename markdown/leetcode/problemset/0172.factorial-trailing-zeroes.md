class Solution {
    fun trailingZeroes(n: Int): Int {
        var now = 5
        var res = 0
        while (now <= n) {
            var t = now
            while (t % 5 == 0) {
                t /= 5
                res++
            }
            now++
        }
        return res
    }
}
