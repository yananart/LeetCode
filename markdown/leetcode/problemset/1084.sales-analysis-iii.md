```mysql
SELECT product_id,
       product_name
FROM (SELECT p.product_id,
             product_name,
             MIN(sale_date) AS first,
             MAX(sale_date) AS last
      FROM Sales AS s
               JOIN Product AS p on p.product_id = s.product_id
      GROUP BY p.product_id) AS temp
WHERE first >= DATE '2019-01-01'
  AND last < DATE '2019-04-01';
```
