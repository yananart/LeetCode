class Solution {
    fun minBitFlips(start: Int, goal: Int): Int {
        if (start < goal) {
            return minBitFlips(goal, start)
        }
        var t = start
        var s = goal
        var res = 0
        while (t != 0) {
            if (t and 1 != s and 1) {
                res++
            }
            t = t shr 1
            s = s shr 1
        }
        return res
    }
}
