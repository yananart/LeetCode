class Solution {
    fun removeNthFromEnd(head: ListNode?, n: Int): ListNode? {
        val res = ArrayList<ListNode>()
        var now = head
        while (now != null) {
            res.add(now)
            now = now.next
        }
        val length = res.size
        if (n == length) {
            return head?.next
        }
        res[length - n - 1].next = res[length - n].next
        return head
    }
}

            length++
            now = now.next

            return head?.next
        }
        now = head
        repeat(length - n - 1) {
            now = now?.next
        }
        now?.next = now?.next?.next
        return head

class Solution {
    fun removeNthFromEnd(head: ListNode?, n: Int): ListNode? {
        var fast = head
        var slow = head
        repeat(n + 1) {
            if (fast == null) {
                return slow?.next
            }
            fast = fast?.next
        }
        while (fast != null) {
            fast = fast?.next
            slow = slow?.next
        }
        slow?.next = slow?.next?.next
        return head
    }
}
