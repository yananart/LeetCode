class Solution {
    fun imageSmoother(img: Array<IntArray>): Array<IntArray> {
        val res = Array(img.size) { IntArray(img[it].size) }
        for (i in img.indices) {
            for (j in img[i].indices) {
                res[i][j] = img.smoother(i, j)
            }
        }
        return res
    }

    private fun Array<IntArray>.smoother(i: Int, j: Int): Int {
        var items = 1
        var all = this[i][j]
        val top = i - 1 >= 0
        val left = j - 1 >= 0
        val bottom = i + 1 < this.size
        val right = j + 1 < this[0].size
        if (top) {
            items++
            all += this[i - 1][j]
            if (left) {
                items++
                all += this[i - 1][j - 1]
            }
            if (right) {
                items++
                all += this[i - 1][j + 1]
            }
        }
        if (bottom) {
            items++
            all += this[i + 1][j]
            if (left) {
                items++
                all += this[i + 1][j - 1]
            }
            if (right) {
                items++
                all += this[i + 1][j + 1]
            }
        }
        if (left) {
            items++
            all += this[i][j - 1]
        }
        if (right) {
            items++
            all += this[i][j + 1]
        }
        return all / items
    }
}
