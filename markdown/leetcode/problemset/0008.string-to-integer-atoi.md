class Solution {
    fun myAtoi(s: String): Int {
        val chars = s.toCharArray()
        val len = chars.size
        // å»é™¤
        var index = 0
        while (index < len && chars[index] == ' ') {
            index++
        }
        if (index == len) {
            return 0
        }
        // åˆ¤æ–­ç¬¦å·ä½
        var sign = 1
        if (chars[index] == '-') {
            index++
            sign = -1
        } else if (chars[index] == '+') {
            index++
        }
        var res = 0
        while (index < len) {
            val ch = chars[index]
            // ä¸€ç›´åˆ¤æ–­å–æ•°å­—
            if (ch < '0' || ch > '9') {
                break
            }
            val tem = ch - '0'
            val last = res
            res = res * 10 + tem
            if (last != res / 10) {
                // æº¢å‡º
                return if (sign > 0) Int.MAX_VALUE else Int.MIN_VALUE
            }
            index++
        }
        return res * sign
    }
}
