import kotlin.math.max

/**
 * @author Yananart
 */
class Solution {
    fun canJump(nums: IntArray): Boolean {
        var fast = 0
        for (i in nums.indices) {
            if (i <= fast) {
                fast = max(fast, i + nums[i])
                if (fast >= nums.size - 1) {
                    return true
                }
            } else {
                return false
            }
        }
        return false
    }
}
