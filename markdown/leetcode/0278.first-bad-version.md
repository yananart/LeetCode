# 278. 第一个错误的版本

- [🔗题目链接](https://leetcode-cn.com/problems/first-bad-version/）

二分查找，中位是bad就把取左边二分，不是bad就取右边二分，注意计算中位时int溢出。

```kotlin
/**
 * @author Yananart
 */
open class VersionControl {
    // 这个类是用来模拟题中给的父类
    private val badVersion = 4
    fun isBadVersion(version: Int) = version >= badVersion
    open fun firstBadVersion(n: Int) = -1
}

class Solution : VersionControl() {
    override fun firstBadVersion(n: Int): Int {
        var left = 1
        var right = n
        while (true) {
            if (left == right) {
                return left
            } else if (left + 1 == right) {
                return if (isBadVersion(left)) left else right
            }
            // 注意可能int会溢出
            val mid = ((left.toLong() + right.toLong()) / 2).toInt()
            if (isBadVersion(mid)) {
                right = mid
            } else {
                left = mid + 1
            }
        }
    }
}
```

第二种写法：

```kotlin
/**
 * @author Yananart
 */
open class VersionControl {
    // 这个类是用来模拟题中给的父类
    private val badVersion = 4
    fun isBadVersion(version: Int) = version >= badVersion
    open fun firstBadVersion(n: Int) = -1
}

class Solution : VersionControl() {
    override fun firstBadVersion(n: Int): Int {
        var left = 1
        var right = n
        while (left < right) {
            val mid = left + (right - left) / 2
            if (isBadVersion(mid)) {
                right = mid
            } else {
                left = mid + 1
            }
        }
        return left
    }
}
```
