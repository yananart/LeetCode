class Solution {
    fun digArtifacts(n: Int, artifacts: Array<IntArray>, dig: Array<IntArray>): Int {
        val area = Array(n) { IntArray(n) { 0 } }
        // æ ‡è®°å‘æ˜ä½ç½®
        for (item in dig) {
            area[item[0]][item[1]] = 1
        }

        var res = 0
        // åˆ¤æ–­
        for (item in artifacts) {
            var find = true
            for (i in item[0]..item[2]) {
                for (j in item[1]..item[3]) {
                    if (area[i][j] != 1) {
                        find = false
                        break
                    }
                }
            }
            if (find) {
                res++
            }
        }
        return res
    }
}
