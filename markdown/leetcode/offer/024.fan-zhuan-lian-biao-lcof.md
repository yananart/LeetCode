class Solution {
    fun reverseList(head: ListNode?): ListNode? {
        var now = head
        var res: ListNode? = null
        while (now != null) {
            val next = now.next
            now.next = res
            res = now
            now = next
        }
        return res
    }
}

class ListNode(var `val`: Int) {
    var next: ListNode? = null
}
